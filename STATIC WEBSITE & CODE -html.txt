function togglePassword(toggle){
    const input = toggle.parentElement.querySelector('input');
    if(!input) return;
    if(input.type === 'password'){
        input.type = 'text';
        toggle.textContent = '🙈';
    } else {
        input.type = 'password';
        toggle.textContent = '👁';
    }
}
function registerUser(){
    const username = document.getElementById('user')?.value?.trim();
    const password = document.getElementById('password')?.value || '';
    const confirmPassword = document.getElementById('confirmPassword')?.value || '';
    if(!username || !password || !confirmPassword){
        alert('Please fill in all fields.');
        return;
    }
    if(password !== confirmPassword){
        alert('Passwords do not match.');
        return;
    }
    document.getElementById('loginOptions').innerHTML = `
        <h2>Account Created</h2>
        <p style="color:#cbd5e1;line-height:1.6;">Your account was created successfully. Redirecting to sign in...</p>`;
    setTimeout(showSignIn, 1200);
}

function submitAddRequest(){
    const name = document.getElementById('requestName')?.value?.trim();
    const date = document.getElementById('requestDate')?.value;
    const type = document.getElementById('requestType')?.value;
    if(!name){
        alert('Please enter the file name.');
        return;
    }
    if(!date){
        alert('Please select a due date.');
        return;
    }
    if(!type){
        alert('Please select a request type.');
        return;
    }
    const formattedDate = new Date(date).toLocaleDateString('en-US');
    const newReq = {name: `${name} (${type})`, status:'PENDING', date: formattedDate};
    requestRecords.push(newReq);
    alert('Request added successfully!');
    showRequests('view');
}
function deleteRequest(){
    if(requestRecords.length === 0){
        alert('No requests available to delete.');
        return;
    }
    requestRecords.pop();
    renderRequestPage(currentRequestMode);
    alert('Last request removed.');
}

function updateRequestStatus(index, newStatus){
    if(index < 0 || index >= requestRecords.length) return;
    if(newStatus === 'REJECTED'){
        const reason = prompt('Enter rejection reason for this request:');
        if(!reason){
            alert('Rejection cancelled. No reason provided.');
            return;
        }
        requestRecords[index].status = 'REJECTED';
        requestRecords[index].reason = reason;
    } else {
        requestRecords[index].status = newStatus;
        if(newStatus === 'APPROVED'){
            delete requestRecords[index].reason;
        }
    }
    renderRequestPage('manage');
}
function deleteRequestByIndex(index){
    if(index < 0 || index >= requestRecords.length) return;
    requestRecords.splice(index, 1);
    renderRequestPage('manage');
}
function archiveRequest(index){
    if(index < 0 || index >= requestRecords.length) return;
    archivedRecords.push(requestRecords.splice(index, 1)[0]);
    renderRequestPage('manage');
}
